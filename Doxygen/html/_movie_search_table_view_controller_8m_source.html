194 #pragma mark - UISearchDisplayController Helper Methods
195 -(void)filterContentForSearchText:(NSString*)searchText scope:(NSString*)scope
196 {
197  self.movieSearch = nil;
198  if (![AFNetworkReachabilityManager sharedManager].reachable)
199  {
200  NSString *message = @"Internet Required To Search";
201  [self.activeTable showTableMessage:message atPosition:self.alertPosition];
202  }
203  else
204  {
205  self.activeTable.tableFooterView = [UIView new];
206  }
207  //we need to use the property self.searchTerm when calling the method to search or when attempting to cancel it
208  //so that we can ensure that the cancel operation uses the same object as the previously called search method,
209  //otherwise if the objects arent identical the cancelling fails and we keep getting search results for every typed letter
210  [NSObject cancelPreviousPerformRequestsWithTarget:self selector:@selector(searchForMovie:) object:self.searchTerm];
211  self.searchTerm = searchText;
212  [self performSelector:@selector(searchForMovie:) withObject:self.searchTerm afterDelay:0.4f];
213 }

215 - (void)searchForMovie:(NSString *)movieName
216 {
217  @weakify(self);
218  [self.activeTable showActivityIndicatorWithCenter:self.alertPosition];
219  [InteractWithAPI searchNYTForMovie:movieName completion:^(NYTMovieSearch *searchResults, NSError *error)
220  {
221  @strongify(self);
222  [self.activeTable hideActivityIndicator];
223  if (error)
224  {
225  [AlertUser showError:error.localizedDescription customTitle:nil];
226  }
227  else if (searchResults)
228  {
229  self.movieSearch = searchResults;
230  self.searchTerm = movieName;
231  [self.activeTable reloadData];
232  }
233  }];
234 }

284 - (void)setupConnectivityCheck
285 {
286  [self.tableView reloadData];
287  //the reload operation is neccessary to ensure that the backround message displayed is correct - it changes if no internet is detected and without the reloadData above on the first launch it always displays the no internet message
288  // Check the reachability status and show an alert if the internet connection is not available// AFNetworkReachabilityStatusUnknown = -1,// break;[[AFNetworkReachabilityManager sharedManager] setReachabilityStatusChangeBlock:^(AFNetworkReachabilityStatus status)
289  {
290  NSLog(@"Reachability: %@", AFStringFromNetworkReachabilityStatus(status));
291 
292  switch (status)
293  {
294  case AFNetworkReachabilityStatusUnknown:
295 
296  NSLog(@"The reachability status is Unknown");
297 
298 
299  case AFNetworkReachabilityStatusNotReachable:
300  NSLog(@"The reachability status is not reachable");
301  [AlertUser showError:[[[NSError new] returnErrorNoInternet] localizedDescription] customTitle:nil];
302  break;
303 
304  case AFNetworkReachabilityStatusReachableViaWWAN:
305 
306  case AFNetworkReachabilityStatusReachableViaWiFi:
307  NSLog(@"The reachability status is reachable via WiFi");
308  [self.tableView reloadData];
309  break;
310 
311  default:
312  break;
313  }
314  }];
315 }
